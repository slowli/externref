33    pub fn replace_functions(&self, module: &mut Module) -> Result<HashSet<FunctionId>, Error> {
34        let (replaced_count, guarded_fns) = self.patched_fns.replace_calls(module)?;

40    #[cfg_attr(feature = "tracing", tracing::instrument(skip_all))]
41    pub fn process_functions(
42        &self,
43        functions: &[Function<'_>],
44        guarded_fns: &HashSet<FunctionId>,
45        module: &mut Module,
46    ) -> Result<(), Error> {
47        // First, resolve function IDs for exports / imports.
48        let function_ids: Result<Vec<_>, _> = functions
49            .iter()
50            .map(|function| Self::function_id(function, module))
51            .collect();
52        let function_ids = function_ids?;
53
54        // Determine which functions return externrefs (only patched imports or exports can
55        // do that).
56        let mut functions_returning_ref = HashSet::new();
57        if let Some(fn_id) = self.patched_fns.get_ref_id() {
58            functions_returning_ref.insert(fn_id);
59        }
60
61        for (function, &fn_id) in functions.iter().zip(&function_ids) {
62            if let Some(fn_id) = fn_id {
63                let type_id = module.funcs.get(fn_id).ty();
64                let results_len = module.types.get(type_id).results().len();
65                let refs = &function.externrefs;
66                if results_len == 1 && refs.is_set(refs.bit_len() - 1) {
67                    functions_returning_ref.insert(fn_id);
68                }
69
70                if let FunctionKind::Import(_) = function.kind {
71                    transform_import(module, function, fn_id)?;
72                }
73            }
74        }
75
76        let functions_by_id = function_ids
77            .into_iter()
78            .zip(functions)
79            .filter_map(|(fn_id, function)| fn_id.map(|fn_id| (fn_id, function)));
80        let functions_by_id: HashMap<_, _> = functions_by_id.collect();
81
82        let local_fn_ids: Vec<_> = module.funcs.iter_local().map(|(id, _)| id).collect();
83        for fn_id in local_fn_ids {
84            if let Some(function) = functions_by_id.get(&fn_id) {
85                Self::transform_export(module, &functions_returning_ref, fn_id, function)?;
86            } else {
87                let can_have_locals = guarded_fns.contains(&fn_id);
88                Self::transform_local_fn(module, &functions_returning_ref, can_have_locals, fn_id)?;
89            }
90        }
91
92        Ok(())
93    }

95    #[cfg_attr(
96        feature = "tracing",
97        tracing::instrument(
98            level = "trace",
99            skip_all,
100            ret, err,
101            fields(kind = ?function.kind, name = function.name)
102        )

104    fn function_id(function: &Function<'_>, module: &Module) -> Result<Option<FunctionId>, Error> {
105        Ok(Some(match function.kind {
106            FunctionKind::Export => {
107                let export = module
108                    .exports
109                    .iter()
110                    .find(|export| export.name == function.name);
111                let export = export.ok_or_else(|| Error::NoExport(function.name.to_owned()))?;
112                match &export.item {
113                    ExportItem::Function(fn_id) => *fn_id,
114                    _ => return Err(Error::UnexpectedExportType(function.name.to_owned())),
115                }
116            }
117
118            FunctionKind::Import(module_name) => {
119                let Some(import_id) = module.imports.find(module_name, function.name) else {
120                    // The function is declared, but not actually used from the module.
121                    // This is fine for us.
122                    return Ok(None);
123                };
124                match module.imports.get(import_id).kind {
125                    ImportKind::Function(fn_id) => fn_id,
126                    _ => {
127                        return Err(Error::UnexpectedImportType {
128                            module: module_name.to_owned(),
129                            name: function.name.to_owned(),
130                        });
131                    }
132                }
133            }
134        }))
135    }

137    #[cfg_attr(
138        feature = "tracing",
139        tracing::instrument(skip_all, err, fields(name = function.name))
140    )]
141    #[allow(clippy::needless_collect)] // false positive
142    fn transform_export(
143        module: &mut Module,
144        functions_returning_ref: &HashSet<FunctionId>,
145        fn_id: FunctionId,
146        function: &Function<'_>,
147    ) -> Result<(), Error> {
148        let local_fn = module.funcs.get_mut(fn_id).kind.unwrap_local_mut();
149        let (params, results) = patch_type_inner(&module.types, function, local_fn.ty())?;
150
151        let mut locals_mapping = HashMap::new();
152        for idx in function.externrefs.set_indices() {
153            if let Some(arg) = local_fn.args.get_mut(idx) {
154                let new_local = module.locals.add(EXTERNREF);
155                locals_mapping.insert(new_local, *arg);
156                *arg = new_local;
157            }
158        }
159        let ref_args: Vec<_> = locals_mapping.keys().copied().collect();
160
161        let mut calls_visitor = RefCallDetector {
162            locals: &mut module.locals,
163            functions_returning_ref,
164            new_locals: HashMap::default(),
165        };
166        ir::dfs_pre_order_mut(&mut calls_visitor, local_fn, local_fn.entry_block());
167        let mut new_locals = calls_visitor.new_locals;
168        new_locals.extend(locals_mapping);
169
170        // Determine which `local.get $arg` instructions must be replaced with new arg locals.
171        let mut locals_visitor = LocalReplacementCounter::new(ref_args.into_iter(), new_locals);
172        ir::dfs_in_order(&mut locals_visitor, local_fn, local_fn.entry_block());
173        let mut replacer = LocalReplacer::from(locals_visitor);
174        // Clone the function with new function types.
175        let mut cloner =
176            FunctionCloner::new(FunctionBuilder::new(&mut module.types, &params, &results));
177        ir::dfs_in_order(&mut cloner, local_fn, local_fn.entry_block());
178        cloner.clone_function(local_fn, &mut replacer);
179
180        Ok(())
181    }
182
183    /// What we want to do here and in [`Self::transform_export()`] is to patch some
184    /// of locals that have the `i32` type, but must have the `externref` type as per
185    /// patched functions. There are two types of such locals:
186    ///
187    /// - `externref` arguments for exports, which we know by collecting function signatures
188    ///   from the custom section
189    /// - Locals assigned from calling a function that returns `externref`. We know such functions
190    ///   in advance; they are among imported functions (in which case whether a function
191    ///   returns an `externref` is determined based on the function sig from the custom section),
192    ///   plus the `exernref::get` function.
193    ///
194    /// Locals of the second type can occur in any local function; thus, we need to scan all
195    /// of them. We scan for these locals by searching tuples of `call $fn` + `local.set $r` /
196    /// `local.tee $r` instructions, where `$fn` is a function returning `externref`.
197    /// Thus, we assume that:
198    ///
199    /// - `call.indirect` is not used to produce `externref`s. This seems to be correct
200    ///   for properly produced modules.
201    /// - A local is assigned immediately after the call. This *looks* reasonable; besides
202    ///   being assigned to a local, an `externref` can only be consumed by a function
203    ///   accepting an `externref` argument. Still, this assumption is somewhat shaky.
204    ///   Further, it doesn't really work with functions returning multiple results.
205    ///
206    /// To eliminate these restrictions with 100% certainty, it would be necessary to symbolically
207    /// evaluate each local function to determine the contents of the operand stack at all times
208    /// (which is a significant part of module validation). Doesn't seem worth the effort right now.
209    ///
210    /// After all `externref` locals are found, we determine uses (via `local.get $ref`) for each
211    /// local, taking into account that a local can be reassigned. For call result locals this
212    /// means that we should introduce a new local for each call to be on the safe side.
213    /// (We could reuse locals in some cases, but this requires more work.) A single use is
214    /// encoded as a tuple (sequence ID, index of `local.get $ref` in the sequence).
215    ///
216    /// Finally, after collecting all uses, we replace locals with the new ones. For exports,
217    /// this process is combined with cloning function code.
218    #[cfg_attr(
219        feature = "tracing",
220        tracing::instrument(level = "trace", skip_all, err, fields(fn_id))
221    )]
222    fn transform_local_fn(
223        module: &mut Module,
224        functions_returning_ref: &HashSet<FunctionId>,
225        can_have_locals: bool,
226        fn_id: FunctionId,
227    ) -> Result<(), Error> {
228        let function = module.funcs.get_mut(fn_id);
229        let local_fn = function.kind.unwrap_local_mut();
230
231        let mut calls_visitor = RefCallDetector {
232            locals: &mut module.locals,
233            functions_returning_ref,
234            new_locals: HashMap::default(),
235        };
236        ir::dfs_pre_order_mut(&mut calls_visitor, local_fn, local_fn.entry_block());
237        let new_locals = calls_visitor.new_locals;
238        if new_locals.is_empty() {
239            #[cfg(feature = "tracing")]
240            tracing::trace!("no new locals; skipping function transform");
241            return Ok(());
242        } else if !can_have_locals {
243            return Err(Error::UnexpectedCall {
244                function_name: function.name.clone(),
245                code_offset: function_offset(local_fn),
246            });
247        }
248
249        #[cfg(feature = "tracing")]
250        tracing::debug!(
251            ?fn_id,
252            new_locals.len = new_locals.len(),
253            "replacing function locals"
254        );
255
256        // Determine which `local.get $arg` instructions must be replaced with new arg locals.
257        let mut locals_visitor = LocalReplacementCounter::new(iter::empty(), new_locals);
258        ir::dfs_in_order(&mut locals_visitor, local_fn, local_fn.entry_block());
259        let mut replacer = LocalReplacer::from(locals_visitor);
260        ir::dfs_pre_order_mut(&mut replacer, local_fn, local_fn.entry_block());
261        Ok(())
262    }
263}

265fn function_offset(local_fn: &LocalFunction) -> Option<u32> {
266    local_fn
267        .block(local_fn.entry_block())
268        .iter()
269        .next()
270        .and_then(|(_, location)| get_offset(*location))
271}

273/// Visitor to detect calls to functions returning `externref`s and create a new ref local
274/// for each call.
275#[derive(Debug)]
276struct RefCallDetector<'a> {
277    locals: &'a mut ModuleLocals,
278    functions_returning_ref: &'a HashSet<FunctionId>,
279    /// Mapping from a new local to the old local.
280    new_locals: HashMap<LocalId, LocalId>,
281}
282
283impl RefCallDetector<'_> {
284    fn returns_ref(&self, instr: &ir::Instr) -> bool {
285        if let ir::Instr::Call(call) = instr {
286            self.functions_returning_ref.contains(&call.func)
287        } else {
288            false
289        }
290    }
291
292    fn replace_local(&mut self, local: &mut LocalId) {
293        let new_local = self.locals.add(EXTERNREF);
294        self.new_locals.insert(new_local, *local);
295        *local = new_local;
296    }
297}
298
299impl ir::VisitorMut for RefCallDetector<'_> {
300    fn start_instr_seq_mut(&mut self, instr_seq: &mut ir::InstrSeq) {
301        let mut ref_on_top_of_stack = false;
302        for (instr, _) in &mut instr_seq.instrs {
303            match instr {
304                ir::Instr::LocalSet(local_set) if ref_on_top_of_stack => {
305                    self.replace_local(&mut local_set.local);
306                    ref_on_top_of_stack = false;
307                }
308                ir::Instr::LocalTee(local_tee) if ref_on_top_of_stack => {
309                    self.replace_local(&mut local_tee.local);
310                }
311                _ => {
312                    ref_on_top_of_stack = self.returns_ref(instr);
313                }
314            }
315        }
316    }
317}

325/// Visitor counting mentions of `externref` locals in patched functions.
326///
327/// It is valid to reassign param locals via `local.set` or `local.tee`
328/// (and Rust frequently does this in practice).
329/// Since we change the local type from `i32` to `externref`, we need to track reassignments,
330/// and not change the local ID after reassignment (since it should retain the old `i32` type).
331#[derive(Debug)]
332struct LocalReplacementCounter {
333    locals: HashMap<LocalId, LocalState>,
334    new_locals: HashMap<LocalId, LocalId>,
335    current_seqs: Vec<ir::InstrSeqId>,
336}
337
338impl LocalReplacementCounter {
339    fn new(ref_args: impl Iterator<Item = LocalId>, new_locals: HashMap<LocalId, LocalId>) -> Self {
340        let mut locals: HashMap<_, _> = new_locals
341            .values()
342            .map(|local_id| (*local_id, LocalState::default()))
343            .collect();
344        for arg in ref_args {
345            let old_local = new_locals[&arg];
346            locals.get_mut(&old_local).unwrap().current_replacement = Some(arg);
347        }
348
349        Self {
350            locals,
351            new_locals,
352            current_seqs: vec![],
353        }
354    }
355
356    fn visit_assignment(&mut self, local: LocalId) {
357        if let Some(state) = self.locals.get_mut(&local) {
358            state.current_replacement = None;
359        } else if let Some(old_local) = self.new_locals.get(&local) {
360            let state = self.locals.get_mut(old_local).unwrap();
361            state.current_replacement = Some(local);
362        }
363    }
364}
365
366impl ir::Visitor<'_> for LocalReplacementCounter {
367    fn start_instr_seq(&mut self, instr_seq: &ir::InstrSeq) {
368        self.current_seqs.push(instr_seq.id());
369    }
370
371    fn end_instr_seq(&mut self, _: &ir::InstrSeq) {
372        self.current_seqs.pop();
373    }
374
375    fn visit_local_get(&mut self, instr: &ir::LocalGet) {
376        let local_id = instr.local;
377        let current_seq = *self.current_seqs.last().unwrap();
378        if let Some(state) = self.locals.get_mut(&local_id) {
379            state
380                .replacements
381                .entry(current_seq)
382                .or_default()
383                .push(state.current_replacement);
384        }
385    }
386
387    fn visit_local_set(&mut self, instr: &ir::LocalSet) {
388        self.visit_assignment(instr.local);
389    }
390
391    fn visit_local_tee(&mut self, instr: &ir::LocalTee) {
392        self.visit_assignment(instr.local);
393    }
394}

396#[derive(Debug)]
397struct LocalReplacer {
398    locals: HashMap<LocalId, LocalState>,
399    current_seqs: Vec<ir::InstrSeqId>,
400}
401
402impl LocalReplacer {
403    fn take_replacement(&mut self, seq: ir::InstrSeqId, local: LocalId) -> Option<LocalId> {
404        if let Some(state) = self.locals.get_mut(&local) {
405            if let Some(replacements) = state.replacements.get_mut(&seq) {
406                return replacements.pop().flatten();
407            }
408        }
409        None
410    }
411}
412
413impl From<LocalReplacementCounter> for LocalReplacer {
414    fn from(counter: LocalReplacementCounter) -> Self {
415        // Reverse all replacements to pop them in `Self::take_replacement()` in proper order.
416        let mut locals = counter.locals;
417        for state in locals.values_mut() {
418            for replacements in state.replacements.values_mut() {
419                replacements.reverse();
420            }
421        }
422
423        Self {
424            locals,
425            current_seqs: vec![],
426        }
427    }
428}
429
430impl ir::VisitorMut for LocalReplacer {
431    fn start_instr_seq_mut(&mut self, instr_seq: &mut ir::InstrSeq) {
432        self.current_seqs.push(instr_seq.id());
433    }
434
435    fn end_instr_seq_mut(&mut self, _: &mut ir::InstrSeq) {
436        self.current_seqs.pop();
437    }
438
439    fn visit_local_get_mut(&mut self, instr: &mut ir::LocalGet) {
440        let seq = *self.current_seqs.last().unwrap();
441        if let Some(replacement) = self.take_replacement(seq, instr.local) {
442            instr.local = replacement;
443        }
444    }
445}

447/// Visitor for function cloning.
448#[derive(Debug)]
449struct FunctionCloner {
450    builder: FunctionBuilder,
451    sequence_mapping: HashMap<ir::InstrSeqId, ir::InstrSeqId>,
452}
453
454impl FunctionCloner {
455    fn new(builder: FunctionBuilder) -> Self {
456        Self {
457            builder,
458            sequence_mapping: HashMap::new(),
459        }
460    }
461
462    fn clone_function(self, local_fn: &mut LocalFunction, replacer: &mut LocalReplacer) {
463        let mut builder = self.builder;
464        // We cannot use `VisitorMut` here because we're switching arenas for `InstrSeqId`s.
465        for (old_id, new_id) in &self.sequence_mapping {
466            let seq = local_fn.block_mut(*old_id);
467            let mut instructions = mem::take(&mut seq.instrs);
468            for (instr, _) in &mut instructions {
469                match instr {
470                    ir::Instr::Block(ir::Block { seq })
471                    | ir::Instr::Loop(ir::Loop { seq })
472                    | ir::Instr::Br(ir::Br { block: seq })
473                    | ir::Instr::BrIf(ir::BrIf { block: seq }) => {
474                        *seq = self.sequence_mapping[seq];
475                    }
476
477                    ir::Instr::IfElse(ir::IfElse {
478                        consequent,
479                        alternative,
480                    }) => {
481                        *consequent = self.sequence_mapping[consequent];
482                        *alternative = self.sequence_mapping[alternative];
483                    }
484                    ir::Instr::BrTable(ir::BrTable { blocks, default }) => {
485                        for block in blocks.iter_mut() {
486                            *block = self.sequence_mapping[block];
487                        }
488                        *default = self.sequence_mapping[default];
489                    }
490
491                    ir::Instr::LocalGet(ir::LocalGet { local }) => {
492                        if let Some(new_local) = replacer.take_replacement(*old_id, *local) {
493                            *local = new_local;
494                        }
495                    }
496
497                    _ => { /* Do nothing */ }
498                }
499            }
500
501            *builder.instr_seq(*new_id).instrs_mut() = instructions;
502        }
503
504        *local_fn.builder_mut() = builder;
505    }
506}
507
508impl ir::Visitor<'_> for FunctionCloner {
509    fn start_instr_seq(&mut self, instr_seq: &ir::InstrSeq) {
510        let new_id = if self.sequence_mapping.is_empty() {
511            // entry block
512            self.builder.func_body().id()
513        } else {
514            self.builder.dangling_instr_seq(instr_seq.ty).id()
515        };
516        self.sequence_mapping.insert(instr_seq.id(), new_id);
517    }
518}

528fn transform_import(

530    function: &Function<'_>,
531    fn_id: FunctionId,
532) -> Result<(), Error> {
533    let imported_fn = module.funcs.get_mut(fn_id).kind.unwrap_import_mut();

535    imported_fn.ty = patched_ty;

537}

539fn patch_type(
540    types: &mut ModuleTypes,
541    function: &Function<'_>,
542    ty: TypeId,
543) -> Result<TypeId, Error> {
544    let (params, results) = patch_type_inner(types, function, ty)?;
545    Ok(types.add(&params, &results))
546}
547
548fn patch_type_inner(
549    types: &ModuleTypes,
550    function: &Function<'_>,
551    ty: TypeId,
552) -> Result<(Vec<ValType>, Vec<ValType>), Error> {
553    let (params, results) = types.params_results(ty);
554    if params.len() + results.len() != function.externrefs.bit_len() {
555        return Err(Error::UnexpectedArity {
556            module: fn_module(&function.kind).map(str::to_owned),
557            name: function.name.to_owned(),
558            expected_arity: function.externrefs.bit_len(),
559            real_arity: params.len() + results.len(),
560        });
561    }
562
563    let mut new_params = params.to_vec();
564    let mut new_results = results.to_vec();
565    for idx in function.externrefs.set_indices() {
566        let placement = if idx < new_params.len() {
567            &mut new_params[idx]
568        } else {
569            &mut new_results[idx - new_params.len()]
570        };
571
572        if *placement != ValType::I32 {
573            return Err(Error::UnexpectedType {
574                module: fn_module(&function.kind).map(str::to_owned),
575                name: function.name.to_owned(),
576                location: if idx < new_params.len() {
577                    Location::Arg(idx)
578                } else {
579                    Location::ReturnType(idx - new_params.len())
580                },
581                real_type: new_params[idx],
582            });
583        }
584        *placement = EXTERNREF;
585    }
586
587    #[cfg(feature = "tracing")]
588    tracing::info!(
589        ?params,
590        ?results,
591        ?new_params,
592        ?new_results,
593        "replaced function signature"
594    );
595    Ok((new_params, new_results))
596}

598fn fn_module<'a>(fn_kind: &FunctionKind<'a>) -> Option<&'a str> {
599    match fn_kind {
600        FunctionKind::Export => None,
601        FunctionKind::Import(module) => Some(*module),
602    }
603}

644            .find_map(|export| (export.name == "test").then_some(export.item));
645        let ExportItem::Function(fn_id) = fn_id.unwrap() else {
646            unreachable!()
647        };
648
649        ProcessingState::transform_local_fn(&mut module, &functions_returning_ref, true, fn_id)
650            .unwrap();
651
652        let ref_locals: Vec<_> = module
653            .locals
654            .iter()
655            .filter(|local| local.ty() == EXTERNREF)
656            .collect();
657        assert_eq!(ref_locals.len(), 1, "{ref_locals:?}");
658        let ref_local_id = ref_locals[0].id();
659
660        let local_fn = module.funcs.get(fn_id).kind.unwrap_local();
661        let mut mentions = LocalMentions::default();
662        ir::dfs_in_order(&mut mentions, local_fn, local_fn.entry_block());
663        assert_eq!(mentions.local_counts[&ref_local_id], 2);
664    }
665
666    #[derive(Debug, Default)]
667    struct LocalMentions {
668        local_counts: HashMap<LocalId, usize>,
669    }
670
671    impl ir::Visitor<'_> for LocalMentions {
672        fn visit_local_id(&mut self, local_id: &LocalId) {
673            *self.local_counts.entry(*local_id).or_default() += 1;
674        }
675    }
